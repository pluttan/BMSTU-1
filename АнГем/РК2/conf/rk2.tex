\subsection{–î–æ–∫–∞–∑–∞—Ç—å —Ç–µ–æ—Ä–µ–º—É –æ —Å–≤—è–∑–∏ —Ä–µ—à–µ–Ω–∏–∏ÃÜ –Ω–µ–æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–∏ÃÜ –∏ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–∏ÃÜ –æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–∏ÃÜ –°–õ–ê–£ –∏ —Ç–µ–æ—Ä–µ–º—É –æ —Å—Ç—Ä—É–∫—Ç—É—Ä–µ –æ–±—â–µ–≥–æ —Ä–µ—à–µ–Ω–∏—è –Ω–µ–æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–∏ÃÜ –°–õ–ê–£.}

\textit {–¢–µ–æ—Ä–µ–º–∞ (–æ —Å–≤—è–∑–∏ —Ä–µ—à–µ–Ω–∏–π –Ω–µ–æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–π –∏ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–π –æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–π –°–õ–ê–£).}

\vspace*{15pt}

{\bf {–î–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ}}

($\Rightarrow$)

\vspace*{15pt}

($\Leftarrow$)  

–ü—É—Å—Ç—å $X^0$ - —Ä–µ—à–µ–Ω–∏–µ –Ω–µ–æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–π –°–õ–ê–£ $AX = B$ (—Ç.–µ. $AX^0 = B$),

–∞ $Y$ - —Ä–µ—à–µ–Ω–∏–µ –æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–π –°–õ–ê–£ $AX = \Theta$ (—Ç.–µ. $AY = \Theta$).

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º $X = X^0 + Y$ –∏ –Ω–∞–π–¥–µ–º AX:

$AX = A(X^0 + Y) = AX^0 + AY = B + \Theta = B$, —Ç.–µ. $X$ - —Ä–µ—à–µ–Ω–∏–µ –Ω–µ–æ–¥–Ω–æ—Ä–æ–¥–Ω–æ–π –°–õ–ê–£ $AX = B$.  

\ep

\subsection{–î–æ–∫–∞–∑–∞—Ç—å —Ç–µ–æ—Ä–µ–º—É –æ –±–∞–∑–∏—Å–Ω–æ–º –º–∏–Ω–æ—Ä–µ.}

    \item –õ—é–±—ã–µ —Å—Ç—Ä–æ–∫–∏ (—Å—Ç–æ–ª–±—Ü—ã) –º–∞—Ç—Ä–∏—Ü—ã $A$, –Ω–µ –≤—Ö–æ–¥—è—â–∏–µ –≤ –±–∞–∑–∏—Å–Ω—ã–π –º–∏–Ω–æ—Ä $M$, —è–≤–ª—è—é—Ç—Å—è –ª–∏–Ω–µ–π–Ω—ã–º–∏ –∫–æ–º–±–∏–Ω–∞—Ü–∏—è–º–∏ –±–∞–∑–∏—Å–Ω—ã—Ö —Å—Ç—Ä–æ–∫ (—Å—Ç–æ–ª–±—Ü–æ–≤).

{\bf{–î–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ (–¥–ª—è —Å—Ç—Ä–æ–∫)}}

\vspace*{15pt}

–ü—É—Å—Ç—å –º–∞—Ç—Ä–∏—Ü–∞ $A = (a_{ij})$ –∏–º–µ–µ—Ç —Ç–∏–ø $m \times n$, –ø—É—Å—Ç—å $RgA = r$ –∏ –ø—É—Å—Ç—å $M$ - –±–∞–∑–∏—Å–Ω—ã–π –º–∏–Ω–æ—Ä –º–∞—Ç—Ä–∏—Ü—ã $A$.

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º —Å—Ç—Ä–æ–∫–∏, –Ω–∞ –∫–æ—Ç–æ—Ä—ã—Ö –ø–æ—Å—Ç—Ä–æ–µ–Ω $M$. –≠—Ç–æ –±–∞–∑–∏—Å–Ω—ã–µ —Å—Ç—Ä–æ–∫–∏ –º–∞—Ç—Ä—Ü–∏—ã $A$.

\begin{enumerate}
    \item –î–æ–∫–∞–∂–µ–º, —á—Ç–æ –±–∞–∑–∏—Å–Ω—ã–µ —Å—Ç—Ä–æ–∫–∏ –ª–∏–Ω–µ–π–Ω–æ –Ω–µ–∑–∞–≤–∏—Å–∏–º—ã.
    
    –ü—É—Å—Ç—å –æ—Ç –ø—Ä–æ—Ç–∏–≤–Ω–æ–≥–æ –æ–Ω–∏ –ª–∏–Ω–µ–π–Ω–æ –∑–∞–≤–∏—Å–∏–º—ã $\overset{\text{–ø–æ –∫—Ä–∏—Ç–µ—Ä–∏—é}}{\Rightarrow}$ —Ö–æ—Ç—è –±—ã –æ–¥–Ω–∞ —Å—Ç—Ä–æ–∫–∞ –∏–∑ –Ω–∏—Ö –≤ –º–∞—Ç—Ä–∏—Ü–µ $A$ —è–≤–ª—è–µ—Ç—Å—è –ª–∏–Ω–µ–π–Ω–æ–π –∫–æ–º–±–∏–Ω–∞—Ü–∏–µ–π –æ—Å—Ç–∞–ª—å–Ω—ã—Ö $\Rightarrow$ –≤ –º–∏–Ω–æ—Ä–µ $M$ —Ö–æ—Ç—è –±—ã –æ–¥–Ω–∞ —Å—Ç—Ä–æ–∫–∞ —è–≤–ª—è–µ—Ç—Å—è –ª–∏–Ω–µ–π–Ω–æ–π –∫–æ–º–±–∏–Ω–∞—Ü–∏–µ–π –æ—Å—Ç–∞–ª—å–Ω—ã—Ö $\overset{\text{–ø–æ —Å–≤-–≤—É det}}{\Rightarrow}$ $detM = 0$,
    
    –ü—Ä–æ—Ç–∏–≤–æ—Ä–µ—á–∏–µ, —Ç.–∫. $M$ - –±–∞–∑–∏—Å–Ω—ã–π –º–∏–Ω–æ—Ä.

    \item –î–æ–∫–∞–∂–µ–º, —á—Ç–æ –ª—é–±–∞—è —Å—Ç—Ä–æ–∫–∞ –º–∞—Ç—Ä–∏—Ü—ã $A$, –Ω–µ –≤—Ö–æ–¥—è—â–∞—è –≤ –±–∞–∑–∏—Å–Ω—ã–π –º–∏–Ω–æ—Ä $M$, —è–≤–ª—è–µ—Ç—Å—è –ª–∏–Ω–µ–π–Ω–æ–π –∫–æ–º–±–∏–Ω–∞—Ü–∏–µ–π –±–∞–∑–∏—Å–Ω—ã—Ö —Å—Ç—Ä–æ–∫.

    –ü—É—Å—Ç—å –±–∞–∑–∏—Å–Ω—ã–π –º–∏–Ω–æ—Ä $M$ —Ä–∞—Å–ø–æ–ª–æ–∂–µ–Ω –≤ –≤–µ—Ä—Ö–Ω–µ–º –ª–µ–≤–æ–º —É–≥–ª—É –º–∞—Ç—Ä–∏—Ü—ã $A$:
    
    %–º–∞—Ç—Ä–∏—Ü–∞ - –º–∏–Ω–æ—Ä
    %     |a(11) ... a(1r)|
    % M = |...       ...  |
    %     |a(r1) ... a(rr)|
    $$
    M = 
    \begin{pmatrix}
        a_{11}&\ldots&a_{1r}\\
        \vdots&\ddots&\vdots\\
        a_{r1}&\ldots&a_{rr}\\
    \end{pmatrix}
    $$
    –î–æ–±–∞–≤–∏–º –∫ $M$ –ª—é–±—É—é $i$-—é {\bf {–Ω–µ –±–∞–∑–∏—Å–Ω—É—é}} —Å—Ç—Ä–æ–∫—É –∏ –ª—é–±–æ–π $j$-–π —Å—Ç–æ–ª–±–µ—Ü (–≤–æ–∑–º–æ–∂–Ω–æ –¥–∞–∂–µ –±–∞–∑–∏—Å–Ω—ã–π):
    
    $$
    \Delta_j = 
    \begin{pmatrix}
        a_{11}&\ldots&a_{1r}&a_{1j}\\
        \vdots&&\vdots&\vdots\\
        a_{r1}&\ldots&a_{rr}&a_{rj}\\
        a_{i1}&\ldots&a_{ir}&a_{ij}\\
    \end{pmatrix}
    $$
–ü–æ—Ä—è–¥–æ–∫ $\Delta_j$ —Ä–∞–≤–µ–Ω $r + 1$, —Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ $\Delta_j = 0.$

–†–∞–∑–ª–æ–∂–∏–º $\Delta_j$ –ø–æ –ø–æ—Å–ª–µ–¥–Ω–µ–º—É —Å—Ç–æ–ª–±—Ü—É:

$\Delta_j = a_{1j}A_{1j} + ... + a_{rj}A_{rj} + a_{ij}A_{ij} = 0$, –≥–¥–µ $A_{kj}$ - —ç—Ç–æ –∞–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–∏–µ –¥–æ–ø–æ–ª–Ω–µ–Ω–∏—è —ç–ª–µ–º–µ–Ω—Ç–æ–≤ $a_{kj}$ –≤ $\Delta_j$.

–ó–∞–º–µ—Ç–∏–º, —á—Ç–æ
\begin{enumerate}
    \item[1)] —ç—Ç–∏ –∞–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–∏–µ –¥–æ–ø–æ–ª–Ω–µ–Ω–∏—è $A_{kj}$ –Ω–µ –∑–∞–≤–∏—Å—è—Ç –æ—Ç –Ω–æ–º–µ—Ä–∞ $j$, —Ç.–∫. –ø—Ä–∏ –∏—Ö –≤—ã—á–∏—Å–ª–µ–Ω–∏–∏ $j$-–π —Å—Ç–æ–ª–±–µ—Ü –≤—ã—á–µ—Ä–∫–∏–≤–∞–µ—Ç—Å—è.
    \item[2)] $A_{ij} = (-1)^{(r + 1) + (r + 1)}M = (-1)^{2(r + 1)}M = M \ne 0,$
\end{enumerate}

–í—ã—Ä–∞–∑–∏–º —ç–ª–µ–º–µ–Ω—Ç $a_{ij}:$

$$a_{ij} = \underbrace{\frac{-A_{1j}}{M}}_{b_1}a_{1j} - ... - \underbrace{\frac{-A_{rj}}{M}}_{b_r}a_{rj}, \text{ —Ç.–µ.}$$ 

$$a_{ij} = b_1a_{1j} + ... + b_ra_{rj}, \text{ –≥–¥–µ } b_1, ..., b_r \text{ –Ω–µ –∑–∞–≤–∏—Å—è—Ç –æ—Ç –Ω–æ–º–µ—Ä–∞ j.}$$

\subsection{–î–æ–∫–∞–∑–∞—Ç—å –∫—Ä–∏—Ç–µ—Ä–∏–∏ÃÜ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è –æ–±—Ä–∞—Ç–Ω–æ–∏ÃÜ –º–∞—Ç—Ä–∏—Ü—ã.}

\textit{–∫—Ä–∏—Ç–µ—Ä–∏–∏ÃÜ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è –æ–±—Ä–∞—Ç–Ω–æ–∏ÃÜ –º–∞—Ç—Ä–∏—Ü—ã}

\vspace*{15pt}

–î–ª—è –∫–≤–∞–¥—Ä–∞—Ç–Ω–æ–π –º–∞—Ç—Ä–∏—Ü—ã $A$ $\exists$ –æ–±—Ä–∞—Ç–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞ $A^{-1}$ $\iff$ $detA \ne 0$ (—Ç.–µ. –∫–æ–≥–¥–∞ $A$ - –Ω–µ–≤—ã—Ä–æ–∂–¥–µ–Ω–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞).

\vspace*{15pt}

{\bf {–î–æ–∫–∞–∑–∞—Ç–µ–ª—å—Å—Ç–≤–æ}}

($\Rightarrow$)  

–ü—É—Å—Ç—å $\exists A^{-1}$. –î–æ–∫–∞–∂–µ–º, —á—Ç–æ $detA \ne 0$.

–ü–æ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—é –æ–±—Ä–∞—Ç–Ω–æ–π –º–∞—Ç—Ä–∏—Ü—ã,

–í–æ–∑—å–º–µ–º $det$ –æ—Ç –ª–µ–≤–æ–π –∏ –ø—Ä–∞–≤–æ–π —á–∞—Å—Ç–∏:
$$det(AA^{-1}) = detE$$

–ü–æ —Å–≤–æ–π—Å—Ç–≤–∞–º $det$:
$$detA\cdot detA^{-1} = 1,$$

–ø—Ä–æ–∏–∑–≤–µ–¥–µ–Ω–∏–µ —á–∏—Å–µ–ª —Ä–∞–≤–Ω–æ 1 $\rightarrow$ $detA \ne 0$, $det(A^{-1}) \ne 0$.

\vspace*{15pt}

($\Leftarrow$)

–ü—É—Å—Ç—å $detA \ne 0$.

\begin{enumerate}
    \item –ü–æ—Å—Ç—Ä–æ–∏–º –º–∞—Ç—Ä–∏—Ü—É $A^{-1}:$
    \begin{enumerate}
        \item[1)] –ù–∞–π–¥–µ–º $\forall$ –∞–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–∏–µ –¥–æ–ø–æ–ª–Ω–µ–Ω–∏—è $A_{ij}$ –∏ —Å–æ—Å—Ç–∞–≤–∏–º –∏–∑ –Ω–∏—Ö –º–∞—Ç—Ä–∏—Ü—É $(A_{ij})$.
        \item[2)] –¢—Ä–∞–Ω—Å–ø–æ–Ω–∏—Ä—É–µ–º –º–∞—Ç—Ä–∏—Ü—É $(A_{ij}):$
    $$(A_{ji}) = (A_{ij})^T$$
        \item[3)] –†–∞—Å—Å–º–æ—Ç—Ä–∏–º –º–∞—Ç—Ä–∏—Ü—É $B = (b_{ij})$, –≥–¥–µ $b_{ij} = \frac{A_{ji}}{detA}$, —Ç.–µ. $B = \frac{1}{detA}(A_{ji})$
    \end{enumerate}
    \item –ü—Ä–æ–≤–µ—Ä–∏–º, —á—Ç–æ –ø–æ—Å—Ç—Ä–æ–µ–Ω–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞ $B$ –∏ –±—É–¥–µ—Ç $A^{-1}$.

    –í —Å–∞–º–æ–º –¥–µ–ª–µ, $B$ - –∫–≤–∞–¥—Ä–∞—Ç–Ω–∞—è –∏ –æ—Å—Ç–∞–ª–æ—Å—å –ø—Ä–æ–≤–µ—Ä–∏—Ç—å, —á—Ç–æ $AB = E$ (–∏ $BA = E$).

    –û–±–æ–∑–Ω–∞—á–∏–º $AB$ —á–µ—Ä–µ–∑ $C = (c_{ik})$

    –ù–∞–π–¥–µ–º $c_{ik} = \overset{n}{\underset{j = 1}{\sum}}a_{ij}b_{jk} = \overset{n}{\underset{j = 1}{\sum}}a_{ij}\frac{A_{kj}}{detA} = \frac{1}{detA}\overset{n}{\underset{j = 1}{\sum}}a_{ij}A_{kj} = \begin{cases}1, i = k\\ 0, i \ne k\end{cases}$ 
    
    —Ç.–∫. ${\underset{j = 1}{\sum}}a_{ij}A_{kj} = \begin{cases}detA, i = k\\ 0, i \ne k \text{ (–ø–æ —Ç. –æ "—Ñ–∞–ª—å—à–∏–≤–æ–º"\ —Ä–∞–∑–ª–æ–∂–µ–Ω–∏–∏ –æ–ø—Ä–µ–¥–µ–ª–∏—Ç–µ–ª—è*)}\end{cases}$

    –°–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ, $C = E$, –∏ $AB = E$.
    –ê–Ω–∞–ª–æ–≥–∏—á–Ω–æ –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç—Å—è, —á—Ç–æ $BA = E$.
    –ò–∑ $1, 2 \Rightarrow$ $B$ —è–≤–ª—è–µ—Ç—Å—è $A^{-1}$ –¥–ª—è $A$.

\end{enumerate}

\ep

\subsection{–î–æ–∫–∞–∑–∞—Ç—å –∫—Ä–∏—Ç–µ—Ä–∏–∏ÃÜ –ö—Ä–æ–Ω–µ–∫–µ—Ä–∞ ‚Äî –ö–∞–ø–µ–ª–ª–∏ —Å–æ–≤–º–µ—Å—Ç–Ω–æ—Å—Ç–∏ –°–õ–ê–£.}

\subsection{–í—ã–≤–µ—Å—Ç–∏ —Ñ–æ—Ä–º—É–ª—ã –ö—Ä–∞–º–µ—Ä–∞ –¥–ª—è —Ä–µ—à–µ–Ω–∏—è —Å–∏—Å—Ç–µ–º—ã –ª–∏–Ω–µ–∏ÃÜ–Ω—ã—Ö —É—Ä–∞–≤–Ω–µ–Ω–∏–∏ÃÜ —Å –æ–±—Ä–∞—Ç–∏–º–æ–∏ÃÜ –º–∞—Ç—Ä–∏—Ü–µ–∏ÃÜ.}
